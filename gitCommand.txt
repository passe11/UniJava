Setup :
	git config --global user.name “[firstname lastname]”
	git config --global user.email “[valid-email]”
Started:
	git init
	git clone
	
Stage/Snapshot:
	git status
	git add  . 
	git reset
	git commit "   "

	git branch
	git branch _____

	git merge
	git log
